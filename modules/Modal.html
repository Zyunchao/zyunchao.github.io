# Modal

模态框组件

将 Element Dialogopen in new window & Element Draweropen in new window 进行了整合封装

组件功能：

  • 整合 dialog + drawer，去除差异化
  • 优化样式
  • 便捷传入按钮
  • 可自定义 slot#title、slot#default、slot#footer
  • drawer 内置 el-scrollbar 滚动条；
  • 可内部统一配置模式化

注意

出于对样式的统一

dialog 的宽度默认为 '70%'；
drawer 的宽度默认为 50%
dialog 顶部距离统一设置为 '5vh'；
dialog-body 最小高度统一设置为 '150px'；

如有不同需求，请覆盖对应的配置

# Drawer 模式

通过给组件传递 type="drawer" 开启 drawer 模式

<template>
    <div class="examples-base-wrapper">
        <el-button type="primary" @click="handleClick">
            打开
        </el-button>
    </div>

    <g-modal v-model="drawerShow" append-to-body title="这是一个抽屉" :btns="btns" type="drawer">
        这是抽屉的内容
    </g-modal>
</template>

<script lang="ts">
export default {
    name: 'DialogTest'
}
</script>

<script lang="ts" setup>
import { toRaw, watch, ref, computed, reactive, toRefs } from 'vue'
import { BtnProps } from '@component/index'

const drawerShow = ref<boolean>(false)

const btns = reactive<BtnProps[]>([
    {
        label: '提交',
        type: 'primary',
        loading: true,
        onClick: () => {
            console.log(`%c 提交`, 'color: #67c23a;')
        }
    },
    {
        label: '保存',
        type: 'primary',
        onClick: () => {
            console.log(`%c 保存`, 'color: #67c23a;')
        }
    }
])

const handleClick = () => {
    drawerShow.value = true
}
</script>

    <g-modal v-model="dialogShow" append-to-body title="这是一个对话框" :btns="btns" type="dialog">

<script lang="ts">
export default {
    name: 'DialogTest'

import { toRaw, watch, ref, computed, reactive, toRefs } from 'vue'

            console.log(`%c 提交`, 'color: #67c23a;')

    {
        label: '保存',
        type: 'primary',
        onClick: () => {
            console.log(`%c 保存`, 'color: #67c23a;')
        }
    }
])

const handleClick = () => {
    dialogShow.value = true
}
</script>

# 去除底部

隐藏底部行为或传递 hideFooter（优先级1），或不传递 btns 配置数组、或 btns 数组长度为0

<template>
    <div class="examples-base-wrapper">
        <el-button type="primary" @click="handleClickDialog">
            打开Dialog
        </el-button>
        <el-button type="primary" @click="handleClickDrawer">
            打开Drawer
        </el-button>
    </div>

    <g-modal
        v-model="dialogShow"
        hide-footer
        append-to-body
        title="这是一个对话框"
        :btns="btns"
        type="dialog"
    >
        这是对话框的内容
    </g-modal>

    <g-modal
        v-model="drawerShow"
        hide-footer
        append-to-body
        title="这是一个抽屉"
        :btns="btns"
        width="70%"
        type="drawer"
    >
        这是抽屉的内容
    </g-modal>
</template>

<script lang="ts">
export default {
    name: 'DialogTest'
}
</script>

<script lang="ts" setup>
import { toRaw, watch, ref, computed, reactive, toRefs } from 'vue'
import { BtnProps } from '@component/index'

const dialogShow = ref<boolean>(false)
const drawerShow = ref<boolean>(false)

const btns = reactive<BtnProps[]>([
    {
        label: '提交',
        type: 'primary',
        loading: true,
        onClick: () => {
            console.log(`%c 提交`, 'color: #67c23a;')
        }
    },
    {
        label: '保存',
        type: 'primary',
        onClick: () => {
            console.log(`%c 保存`, 'color: #67c23a;')
        }
    }
])

const handleClickDialog = () => {
    dialogShow.value = true
}

const handleClickDrawer = () => {
    drawerShow.value = true
}
</script>
